1. 가장 성능좋은 cpu
SELECT cpu_name
FROM cpu 
WHERE clock_frequency=(SELECT MAX(clock_frequency) FROM cpu WHERE clock_frequency)
      AND
      l2_cache=(SELECT MAX(l2_cache) FROM cpu WHERE l2_cache);
2. 가장 성능좋은 cpu와 호환되는 가장 좋은 mainboard
SELECT mb_name, MAX(price)
FROM mainboard 
WHERE	cpu_socket=(SELECT cpu_socket 
		FROM cpu 
		WHERE clock_frequency=(SELECT MAX(clock_frequency) FROM cpu WHERE clock_frequency)
			AND
			l2_cache=(SELECT MAX(l2_cache) FROM cpu WHERE l2_cache)
		);
