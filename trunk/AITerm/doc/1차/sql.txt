템플릿6_호환0
SELECT cpu_name, mb_name, cpu.price, mainboard.price FROM cpu, mainboard WHERE 
cpu.price + mainboard.price < 100000
ORDER BY cpu.clock_frequency DESC LIMIT 1

템플릿6_호환1
SELECT cpu_name, mb_name
FROM cpu, mainboard
WHERE cpu.cpu_socket = mainboard.cpu_socket AND
      clock_frequency+l2_cache = (SELECT MAX(clock_frequency+l2_cache) FROM cpu )
ORDER BY mainboard.price DESC LIMIT 1

FROM cpu
WHERE cpu.clock_frequency / cpu.price > 0
ORDER BY cpu.clock_frequency / cpu.price DESC LIMIT 1
